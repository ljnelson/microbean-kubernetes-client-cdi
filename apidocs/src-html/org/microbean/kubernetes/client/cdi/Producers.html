017package org.microbean.kubernetes.client.cdi;
018
019import javax.enterprise.context.ApplicationScoped;
020
021import javax.enterprise.inject.Produces;
022
023import io.fabric8.kubernetes.client.Config;
024import io.fabric8.kubernetes.client.ConfigBuilder;
025import io.fabric8.kubernetes.client.DefaultKubernetesClient;
026import io.fabric8.kubernetes.client.KubernetesClient;
027
028import io.fabric8.kubernetes.client.utils.HttpClientUtils;
029
030import okhttp3.OkHttpClient;
031
032@ApplicationScoped
033class Producers {
034
035  private Producers() {
036    super();
037  }
038
039  @Produces
040  private static final OkHttpClient produceOkHttpClient(final Config config) {
041    return HttpClientUtils.createHttpClient(config);
042  }
043
044  @Produces
045  private static final Config produceConfig() {
046    return new ConfigBuilder().build();
047  }
048  
049  @Produces
050  private static final KubernetesClient produceKubernetesClient(final OkHttpClient httpClient, final Config config) {
051
052    /*
053      Note that the kubernetes-client project is entirely
054      undocumented, and the documentation that happens to exist is
055      incorrect.
056
057      To create a
058      self-configuring-from-System-properties-and-environment-variables
059      KubernetesClient, you just create a new DefaultKubernetesClient.
060
061      You can use an io.fabric8.kubernetes.client.ConfigBuilder, which
062      appears to be generated by some process as there is no source
063      code for it, and call these builder methods (I figured out what
064      they were by running javap).  In general, the "write" property
065      is withXYZ(), and the "read" property is the concatenation of
066      "get" or "is" and XYZ.
067
068      public ClientBuilder withTrustCerts(boolean);
069      public ClientBuilder withMasterUrl(java.lang.String);
070      public ClientBuilder withApiVersion(java.lang.String);
071      public ClientBuilder withNamespace(java.lang.String);
072      public ClientBuilder withCaCertFile(java.lang.String);
073      public ClientBuilder withCaCertData(java.lang.String);
074      public ClientBuilder withClientCertFile(java.lang.String);
075      public ClientBuilder withClientCertData(java.lang.String);
076      public ClientBuilder withClientKeyFile(java.lang.String);
077      public ClientBuilder withClientKeyData(java.lang.String);
078      public ClientBuilder withClientKeyAlgo(java.lang.String);
079      public ClientBuilder withClientKeyPassphrase(java.lang.String);
080      public ClientBuilder withUsername(java.lang.String);
081      public ClientBuilder withPassword(java.lang.String);
082      public ClientBuilder withOauthToken(java.lang.String);
083      public ClientBuilder withWatchReconnectInterval(int);
084      public ClientBuilder withWatchReconnectLimit(int);
085      public ClientBuilder withConnectionTimeout(int);
086      public ClientBuilder withRequestTimeout(int);
087      public ClientBuilder withRollingTimeout(long);
088      public ClientBuilder withScaleTimeout(long);
089      public ClientBuilder withLoggingInterval(int);
090      public ClientBuilder withWebsocketTimeout(long);
091      public ClientBuilder withWebsocketPingInterval(long);
092      public ClientBuilder withHttpProxy(java.lang.String);
093      public ClientBuilder withHttpsProxy(java.lang.String);
094      public ClientBuilder withProxyUsername(java.lang.String);
095      public ClientBuilder withProxyPassword(java.lang.String);
096      public ClientBuilder withNoProxy(java.lang.String...);
097      public ClientBuilder addToNoProxy(java.lang.String...);
098      public ClientBuilder removeFromNoProxy(java.lang.String...);
099      public ClientBuilder withUserAgent(java.lang.String);
100      public ClientBuilder withTlsVersions(okhttp3.TlsVersion...);
101      public ClientBuilder addToTlsVersions(okhttp3.TlsVersion...);
102      public ClientBuilder removeFromTlsVersions(okhttp3.TlsVersion...);
103      public ClientBuilder addToErrorMessages(java.lang.Integer, java.lang.String);
104      public ClientBuilder addToErrorMessages(java.util.Map<java.lang.Integer, java.lang.String>);
105      public ClientBuilder removeFromErrorMessages(java.lang.Integer);
106      public ClientBuilder removeFromErrorMessages(java.util.Map<java.lang.Integer, java.lang.String>);
107      public ClientBuilder withErrorMessages(java.util.Map<java.lang.Integer, java.lang.String>);
108    */
109    
110    return new DefaultKubernetesClient(httpClient, config);
111  }
